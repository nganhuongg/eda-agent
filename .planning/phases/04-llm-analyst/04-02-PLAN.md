    def analyze_column(
        state: AgentState,
        column: str,
        rejected_claims: List[str] | None = None,
    ) -> AnalystDecision:
        """Public entry point. Phase 5 wraps with functools.partial for run_loop.

        """
        rejected_claims = rejected_claims or []

                )
